cd %USERPROFILE%\Desktop

:: Create subdirectories in CSC2244
cd CSC2244
mkdir practical theory "exam papers"

:: Create files in each subdirectory
cd practical
echo. > practical.txt
echo. > practical.docx
echo. > practical.pptx
cd ..

cd theory
echo. > theory.txt
echo. > theory.docx
echo. > theory.pptx
cd ..

cd "exam papers"
echo. > exam.txt
echo. > exam.docx
echo. > exam.pptx
cd %USERPROFILE%\Desktop

:: Create Excel files on Desktop
echo. > "Icae Marks.xlsx"

move "Icae Marks.xlsx" Marks
move "Final Exam Marks.xlsx" Marks

xcopy /E /I Marks Exam\Marks
